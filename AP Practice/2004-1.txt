public class WordsList {
	private ArrayList myList;
	public int numWordsOfLength(int len) {
		int count = 0;
		for(int i = 0; i < myList.size(); i++) {
			if(myList[i].length() == len) {
				count++;
			}
		}
		return count;
	}

	public void removeWordsOfLength(int len) {
		for(int i = 0; i < myList.size(); i++) {
			if(myList[i].length() == len) {
				myList[i].remove;
			}
		}
	} 
}
